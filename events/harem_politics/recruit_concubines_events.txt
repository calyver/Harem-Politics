## For dukes ##
bride_show.2000 = {
	hidden = yes
	trigger = {
		is_alive = yes
		is_landed_or_landless_administrative = yes
		hp_accepts_concubinage = yes
		trigger_if = {
			limit = {
				has_variable = jarya_recruited
			}
			var:jarya_recruited < 4
		}
		trigger_if = {
			limit = {
				exists = capital_county
			}
			NOT = {
				capital_county = { has_county_modifier = isolate_capital_decision_modifier }
			}
		}
		NOR = {
			has_character_modifier = isolating_modifier
		}
	}

	immediate = {
		if = {
			limit = {
				NOT = { exists = scope:liege }
			}
			save_scope_as = liege
		}
		if = {
			limit = {
				NOT = { exists = scope:harem_liege }
			}
			save_scope_as = harem_liege
		}
		if = {
			limit = {
				any_in_list = {
					list = pool_candidates
					recruit_concubines_viable_candidate_trigger = {
						OVERLORD = scope:liege
					}
					count = 0
				}
			}
			calyver_create_candidate_effect = yes
			if = {
				limit = {
					exists = scope:bride
				}
				scope:bride = {
					add_to_list = pool_candidates
				}
				clear_saved_scope = bride
			}
		}
		if = {
			limit = {
				any_in_list = {
					list = pool_candidates
					recruit_concubines_viable_candidate_trigger = {
						OVERLORD = scope:liege
					}
					count = 1
				}
			}
			random_in_list = {
				list = pool_candidates
				limit = {
					recruit_concubines_viable_candidate_trigger = {
						OVERLORD = scope:liege
					}
				}
				save_scope_as = jarya
			}
		}
		else = {
			random_in_list = {
				list = pool_candidates
				limit = {
					recruit_concubines_viable_candidate_trigger = {
						OVERLORD = scope:liege
					}
				}
				weight = {
					base = 50
					modifier = {
						exists = court_owner
						add = -25
					}
					modifier = {
						scope:liege.religion = religion:islam_religion
						location ?= {
							county.holder ?= {
								religion != religion:islam_religion
							}
						}
						add = 20
					}
					modifier = { # Christians were especially captured
						scope:liege.religion = religion:islam_religion
						religion = religion:christianity_religion
						add = 30
					}
					modifier = { # Foreigners were more exotic
						culture != scope:liege.culture
						scope:liege.religion = religion:islam_religion
						add = 5
					}
					modifier = { # Locals are preferred
						culture = scope:liege.culture
						scope:liege.religion != religion:islam_religion
						add = 5
					}
					modifier = {
						is_lowborn = no
						add = -25
					}
					# Virginity was prized
					modifier = {
						is_virgin_trigger = yes
						add = 25
					}
					# Skills
					modifier = {
						add = {
							value = stewardship
							add = diplomacy
							add = learning
						}
					}
					# Good genetics
					modifier = { #Trait considerations
						num_of_good_genetic_traits >= 1
						add = {
							value = num_of_good_genetic_traits
							multiply = 10
						}
					}
					# No bad genetic traits
					modifier = {
						num_of_bad_genetic_traits > 0
						add = {
							value = num_of_bad_genetic_traits
							multiply = -5
						}
					}
					# Docile captives would be easier to send
					modifier = {
						has_trait = temperate
						add = 10
					}
					modifier = {
						has_trait = craven
						add = 10
					}
					modifier = {
						has_trait = content
						add = 10
					}
					modifier = {
						has_trait = humble
						add = 10
					}
					modifier = {
						has_trait = lustful
						add = 10
					}
					modifier = {
						has_trait = chaste
						add = -10
					}
					modifier = {
						has_trait = zealous
						add = -20
					}
					modifier = {
						has_trait = strong
						add = 5
					}
					# Poets
					modifier = {
						has_trait = lifestyle_poet
						add = 5
					}
					# Skills
					modifier = { #Intrigue
						add = {
							value = intrigue
							divide = 2
						}
					}
					# Health
					modifier = {
						is_healthy = yes
						add = 25
					}
					modifier = {
						bride_show_beauty_important_parameters_trigger = yes
						add = -10
					}		
				}
				save_scope_as = jarya
			}
		}
		if = {
			limit = {
				scope:jarya.court_owner ?= {
					NOT = { this = scope:liege }
					is_ai = no
				}
				scope:jarya = {
					has_any_court_position = yes
				}
			}
			if = {
				limit = {
					scope:liege.faith.religion = religion:islam_religion
				}
				scope:jarya.court_owner = {
					send_interface_toast = {
						type = event_generic_bad
						title = bride_show.2000.t.court_owner.abducted
						right_icon = scope:liege
						left_icon = scope:jarya
						custom_tooltip = bride_show.2000.desc.court_owner.abducted
					}
				}
			}
			else = {
				scope:jarya.court_owner = {
					send_interface_toast = {
						type = event_generic_neutral
						title = bride_show.2000.t.court_owner.recruited
						right_icon = scope:liege
						left_icon = scope:jarya
						show_as_tooltip = {
							scope:liege = {
								add_courtier = scope:jarya
							}
						}
					}
				}
			}
		}
		scope:jarya = {
			if = {
				limit = {
					is_betrothed = yes
					betrothed ?= { this != scope:liege }
				}
				scope:liege = {
					save_scope_as = actor
				}
				betrothed = { trigger_event = marriage_interaction.0041 }
				clear_saved_scope = actor
				break_betrothal = betrothed
			}
		}
		send_interface_toast = {
			type = event_generic_neutral
			title = bride_show.2000.t
			right_icon = scope:jarya
			add_courtier = scope:jarya
			scope:jarya = {
				if = {
					limit = {
						scope:liege.faith.religion = religion:islam_religion
					}
					add_opinion = {
						target = scope:liege
						modifier = upset_opinion
						opinion = -10
					}
					stress_impact = {
						base = minor_stress_impact_gain
					}
					if = {
						limit = {
							any_close_family_member = {
								NOT = { this = scope:liege }
								count >= 1
							}
						}
						every_close_family_member = {
							limit = {
								NOT = { this = scope:liege }
							}
							add_opinion = {
								target = scope:liege
								modifier = abducted_close_family_crime
							}
							if = {
								limit = {
									is_ai = no
								}
								send_interface_toast = {
									type = event_generic_bad
									title = bride_show.2000.t.close_family
									right_icon = scope:liege
									left_icon = scope:jarya
									custom_tooltip = bride_show.2000.desc.court_owner.abducted
								}
							}
						}
					}
				}
				add_character_flag = {
					flag = recently_abducted_concubine
					years = 2
				}
				#In case they received a marriage proposal already
				# add_character_flag = {
				# 	flag = has_been_offered_as_concubine
				# 	days = 1
				# }
				remove_from_list = pool_candidates
				add_character_flag = {
					flag = blocked_from_leaving
					years = 5
				}
				hidden_effect = {
					return_to_court = yes
					# Chance of conversion to liege's faith
					if = {
						limit = {
							NOR = {
								has_trait = zealous 
								faith = scope:liege.faith
							}
						}
						random = {
							chance = 30
							# Based on current faith
							modifier = {
								faith = {
									faith_hostility_level = {
										target = scope:liege.faith
										value = faith_fully_accepted_level
									}
								}
								add = 30
							}
							modifier = {
								faith = {
									faith_hostility_level = {
										target = scope:liege.faith
										value >= faith_astray_level
									}
								}
								add = -15
							}
							modifier = {
								faith = {
									faith_hostility_level = {
										target = scope:liege.faith
										value >= faith_hostile_level
									}
								}
								add = -15
							}
							modifier = {
								faith = {
									faith_hostility_level = {
										target = scope:liege.faith
										value >= faith_evil_level
									}
								}
								add = -15
							}
							modifier = {
								hp_accepts_harems = yes
								add = 10
							}
							# Based on traits
							modifier = {
								has_trait = cynical
								add = 20
							}
							modifier = {
								OR = {
									has_trait = ambitious
									has_trait = greedy
									has_trait = arrogant
								}
								add = 20
							}
							modifier = {
								has_trait = craven
								add = 20
							}
							# Younger characters are more likely to change faiths
							modifier = {
								age < 20
								add = 15
							}
							# Zealous rulers are more likely to recruit converted concubines
							modifier = {
								scope:liege = { has_trait = zealous }
								add = 15
							}
							# Cynical rulers don't care much about concubines' faith
							modifier = {
								scope:liege = { has_trait = cynical }
								add = -15
							}
							if = {
								limit = {
									scope:liege.religion = religion:islam_religion
									religion != religion:islam_religion
								}
								add_character_flag = {
									flag = abducted_non_believer
									years = 5
								}
							}
							set_character_faith = scope:liege.faith
						}
					}
					# They should be lowborn
					if = {
						limit = {
							house ?= {
								any_house_member = {
									count = 1
								}
							}
						}
						set_to_lowborn = yes
					}
				}
				# remove_character_flag = has_been_offered_as_concubine
			}
		}
		if = {
			limit = {
				NOT = {
					has_variable = jarya_recruited
				}
			}
			set_variable = {
				name = jarya_recruited
				value = 1
				months = 6
			}
			if = {
				limit = {
					OR = {
						highest_held_title_tier = tier_duchy
						AND = {
							highest_held_title_tier = tier_county
							is_ai = no
						}
					}
				}
				random = {
					chance = 50
					modifier = {
						has_trait = lustful
						add = 10
					}
					trigger_event = {
						id = bride_show.2000
						days = { 7 10 }
					}
				}
			}
			else_if = {
				limit = {
					highest_held_title_tier > tier_duchy
				}
				trigger_event = {
					id = bride_show.2000
					days = { 7 10 }
				}
			}
		}
		else = {
			change_variable = {
				name = jarya_recruited
				add = 1
			}
			# Should we repeat the event?
			# For dukes, the maximum is 3
			if = {
				limit = {
					highest_held_title_tier = tier_duchy
				}
				if = {
					limit = {
						var:jarya_recruited < 3
					}
					random = {
						chance = 25
						modifier = {
							has_trait = lustful
							add = 10
						}
						trigger_event = {
							id = bride_show.2000
							days = { 7 10 }
						}
					}
				}
				else = {
					remove_variable = jarya_recruited
				}
			}
			# For kings+, the maximum is 4
			if = {
				limit = {
					highest_held_title_tier > tier_duchy
				}
				if = {
					limit = {
						var:jarya_recruited < 3
					}
					random = {
						chance = 50
						modifier = {
							has_trait = lustful
							add = 10
						}
						trigger_event = {
							id = bride_show.2000
							days = { 7 10 }
						}
					}
				}
				else_if = {
					limit = {
						var:jarya_recruited = 3
					}
					random = {
						chance = 25
						modifier = {
							has_trait = lustful
							add = 10
						}
						modifier = {
							highest_held_title_tier > tier_kingdom
							add = 10
						}
						trigger_event = {
							id = bride_show.2000
							days = { 7 10 }
						}
					}
				}
				else_if = {
					limit = {
						var:jarya_recruited > 3	
					}
					remove_variable = jarya_recruited
				}
			}
		}
		# The AI is a bit slow and will sometimes convert them to Islam before taking them as concubines
		if = {
			limit = {
				is_ai = yes
				is_character_interaction_valid = {
					recipient = scope:jarya
					interaction = make_concubine_interaction
				}
				allowed_more_concubines = yes
			}
			random = {
				chance = 10
				modifier = {
					number_of_concubines < number_of_desired_concubines
					add = 60
				}
				modifier = {
					scope:jarya = {
						num_of_good_genetic_traits > 0
					}
					add = 25
				}
				modifier = {
					scope:jarya = {
						num_of_bad_genetic_traits > 0
					}
					add = -50
				}
				modifier = {
					scope:jarya = {
						can_have_children_with = { CHARACTER = scope:liege } # Only if you can have kids since this is about fertility and genetic traits
						fertility > 0.2
					}
					add = 10
				}
				modifier = {
					is_attracted_to_gender_of = scope:jarya
					add = 10
				}
				modifier = {
					NOT = { is_attracted_to_gender_of = scope:jarya }
					add = -25
				}
				modifier = {
					OR = { 
						has_relation_lover = scope:jarya
						has_relation_friend = scope:jarya
						has_relation_soulmate = scope:jarya
						has_relation_best_friend = scope:jarya
					}
					add = 60
				}
				run_interaction = {
					interaction = make_concubine_interaction
					actor = scope:liege
					secondary_actor = scope:liege
					recipient = scope:jarya
					secondary_recipient = scope:jarya
					execute_threshold = accept
				}
			}
		}
	}
}

		}

		}

			}

##Managing the eligible subjects##
scripted_trigger is_valid_candidate_for_recruitment_trigger = {
	royal_concubines_eligible_candidate_trigger = {
		OVERLORD = $OVERLORD$
	}
	trigger_if = {
		limit = {
			exists = court_owner
		}
		court_owner = {
			harem_politics_valid_bride_show_subject_trigger = {
				OVERLORD = $OVERLORD$
			}
		}
	}
}
scripted_trigger recruit_concubines_eligible_subject_trigger = {
	harem_politics_valid_bride_show_subject_trigger = {
		OVERLORD = scope:liege
	}
	OR = {
		trigger_if = {
			limit = {
				gold >= minor_gold_value
			}
			OR = {
				any_neighboring_and_across_water_top_liege_realm_owner = {
					any_realm_province = {
						recruit_royal_concubines_viable_county_trigger = {
							OVERLORD = scope:liege
						}
					}
				}
				any_realm_province = {
					recruit_royal_concubines_viable_county_trigger = {
						OVERLORD = scope:liege
					}
				}
			}
		}
		any_courtier = {
			royal_concubines_eligible_candidate_trigger = {
