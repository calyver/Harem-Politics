pregnancy.4001 = {
	hidden = yes
	immediate = {
		calyver_first_trimester_pregnancy_effects = yes

			limit = {
				is_trimester_1_trigger = yes
			}
			# Since that was the first month, we have to trigger the roll again every week
			trigger_event = {
				event = pregnancy.4001
				days = 7
			}
		}
		else = {
			trigger_event = {
				event = pregnancy.4002
				days = 7

		}
