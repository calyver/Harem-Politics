#Events for birth on actions

# namespace = birth

birth.1180 = {
	type = character_event
	hidden = yes

	trigger = {
		any_parent = {
			culture = {
				has_cultural_parameter = dishu_system
			}
			is_landed_or_landless_administrative = yes
			can_have_imperial_consorts_trigger = yes
			trigger_if = {
				limit = {
					has_ach_dlc_trigger = yes
					NOT = { has_game_rule = coronation_laws_off }
				}
				NOT = { has_realm_law = uncrowned } # Must be coronated
			}
			exists = primary_spouse
			save_temporary_scope_as = royal_parent
		}
		any_parent = {
			is_spouse_of = scope:royal_parent
			this = scope:royal_parent.primary_spouse
		}
	}

	immediate = {
		if = {
			limit = {
				has_trait = born_in_the_purple
			}
			remove_trait = born_in_the_purple
		}
		add_trait = di_child 
	}
}
