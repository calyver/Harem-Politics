scripted_trigger valid_harem_consort_trigger = {
	is_courtier_of = root
	is_physically_able_ai_adult = yes
	is_available_healthy_adult = yes
	NOR = { 
		has_character_flag = visited_consort_flag 
		has_trait = ill
	}
	OR = {
		AND = {
			is_travelling = yes
			current_travel_plan ={
				travel_plan_owner = root
			}
		}
		AND = {
			is_travelling = no
			root = {
				is_travelling = no
			}
		}
	}
}
