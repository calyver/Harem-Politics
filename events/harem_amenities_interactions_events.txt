harem_amenities_interactions.001 = {
	type = letter_event
	hidden = yes
	title = harem_amenities_interactions.001
	opening = harem_amenities_interactions.001
	desc = harem_amenities_interactions.001
	sender = scope:actor

	immediate = {
		add_character_flag = harem_amenities_completion_1
		remove_character_flag = harem_amenities_1
        set_amenity_level = { type = harem_heated_flooring value = 2 }
		send_interface_message = {
			type = good_omen
			title = harem_amenities_interactions.001_title
			desc = harem_amenities_interactions.001_desc
		}
	}

	option = {
	}
}
