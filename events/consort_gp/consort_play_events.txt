scripted_effect clear_harem_interactions_flags_effect = {
	if = {
		limit = { 
			has_character_flag = is_visiting_harem_character_flag
		}
		remove_character_flag = is_visiting_harem_character_flag
	}
	if = {
		limit = { 
			has_character_flag = is_being_visited_harem_character_flag
		}
		remove_character_flag = is_being_visited_harem_character_flag
	}
	if = {
		limit = { 
			has_character_flag = flag_was_rejected_by_liege
		}
		remove_character_flag = flag_was_rejected_by_liege
	}
	if = {
		limit = { 
			has_character_flag = flag_was_rejected_by_authority
		}
		remove_character_flag = flag_was_rejected_by_authority
	}
	if = {
		limit = { 
			has_character_flag = flag_was_rejected_by_fellow_consort
		}
		remove_character_flag = flag_was_rejected_by_fellow_consort
	}
	if = {
		limit = { 
			has_character_flag = flag_was_rejected_by_subordinate
		}
		remove_character_flag = flag_was_rejected_by_subordinate
	}
}

scripted_trigger can_be_promoted_trigger = {

				can_appoint_char_to_court_position = {
					CHAR = scope:consort
					COURT_POS = grand_consort_court_position
				}
			}
		}
	}
}

scripted_trigger valid_relative_promotion_trigger = {
	any_liege_or_above = {
		this = scope:harem_liege
	}
	is_available_healthy_ai_adult = yes

	NOR = {
		has_trait = ill
		has_trait = wounded
		has_character_modifier = recently_tortured
		any_consort = {
			this = scope:harem_liege
		}
		is_child_of = scope:harem_liege
		any_heir_title = {} # Not heir to anything
		has_hook = scope:harem_liege
	}
	trigger_if = {
		limit = { is_female = yes }
		scope:harem_liege = { faith_dominant_gender_female_or_equal = yes }
	}
	trigger_else = { faith_dominant_gender_male_or_equal = yes }

	#Unlanded or lowly vassal
	trigger_if = {
		limit = {
			is_landed = yes
		}
		OR = {	
			AND = { #Promote count to duke
				highest_held_title_tier = tier_county
				any_held_title = { #Not many titles
					tier = tier_county
					count <= 2
				}
				primary_title = {
					de_jure_liege = {
						OR = {
							holder ?= {
								this = scope:harem_liege
								any_held_title = {
									tier = tier_duchy
									count >= 3
								}
							}
							AND = {
								is_title_created = no
								any_in_de_jure_hierarchy = {
									filter = {
										tier = tier_county
									}
									holder = scope:harem_liege
									percent >= 0.50
								}
							}
						}
						NOT = {
							any_de_jure_county = {
								this = scope:harem_liege.capital_county
							}
						}
					}
				}	
				scope:harem_liege = {
					gold >= create_duchy_title_value_harem_politics
				}
			}
			AND = {	#Promote baron to count
				highest_held_title_tier = tier_barony
				primary_title = {
					de_jure_liege = {
						NOT = {
							this = scope:harem_liege.capital_county
						}
					}
				}	
				scope:harem_liege = {
					any_held_title = {
						tier = tier_county
						count >= 4
					}
				}
			}
			AND = {	#Promote to Noble Family
				highest_held_title_tier <= tier_county	
				scope:harem_liege = {
					government_allows = administrative
					highest_held_title_tier >= tier_kingdom
					trigger_if = {
						limit = {
							highest_held_title_tier = tier_kingdom
						}
						is_independent_ruler = yes
					}
					gold >= 400
				}
			}
		}
		NOR = {
			government_has_flag = government_is_republic
			government_has_flag = government_is_theocracy
		}
	}
	trigger_else = {
		any_held_title = { count = 0 }
	}
	OR = {
		AND = {
			exists = dynasty
			exists = scope:harem_liege.dynasty
			NOT = {
				dynasty = scope:harem_liege.dynasty
			}
		}
		NOT = {
			exists = dynasty
		}
	}
}

scripted_trigger harem_politics_valid_consort_landed_relative_trigger = {
	is_physically_able_ai_adult = yes
	is_travelling = no
	is_at_war = no
	
	#accepts_harems = yes Check

	exists = scope:harem_liege.dynasty
	dynasty ?= {
		NOT = {
			this = scope:harem_liege.dynasty
		}
	}
	NOR = {
		has_relation_rival = scope:harem_liege
		has_relation_nemesis = scope:harem_liege

		has_hook = scope:harem_liege
	}

	scope:harem_liege = {
		NOT = {
			has_hook = prev
		}
	}

	OR = {	
		AND = { #Promote a count to duke
			highest_held_title_tier = tier_county
			any_held_title = { #Not many titles
				tier = tier_county
				count <= 3
			}
			primary_title = {
				de_jure_liege = {
					OR = {
						AND = {
							is_title_created = yes
							holder ?= {
								this = scope:harem_liege
								any_held_title = { #Check. This doesn't matter if the duchy is not created
									tier = tier_duchy
									count >= 3
								}
							}
						}
						AND = {
							is_title_created = no
							any_in_de_jure_hierarchy = {
								filter = {
									tier = tier_county
								}
								holder = scope:harem_liege
								percent >= 0.50
							}
						}
					}
					NOT = { #Check. Can't I reuse the any_in_de_jure_hierarchy?
						any_de_jure_county = {
							this = scope:harem_liege.capital_county
						}
					}
				}
			}	
			scope:harem_liege = {
				gold >= create_duchy_title_value_harem_politics #Check
				highest_held_title_tier >= tier_kingdom
			}
		}
		AND = {	#Promote a baron to count
			highest_held_title_tier = tier_barony
			primary_title = {
				de_jure_liege = {
					NOT = {
						this = scope:harem_liege.capital_county
					}
				}
			}	
			scope:harem_liege = {
				any_held_title = {
					tier = tier_county
					count >= 4
				}
				highest_held_title_tier >= tier_duchy
			}
		}
		#Promote unlanded to landed
		AND = { #Is untitled
			any_held_title = { count = 0 }
			any_heir_title = {} # Not heir to anything
			trigger_if = {
				limit = { is_female = yes }
				$LIEGE$ = { faith_dominant_gender_female_or_equal = yes }
			}
			trigger_else = {
				$LIEGE$ = { faith_dominant_gender_male_or_equal = yes }
			}
		}
		#Promote lowborn to noble. Must be unlanded
		is_lowborn = yes
		AND = { #Give them a Noble Family title
			scope:harem_liege = {
				is_landed = yes
				government_allows = administrative
				highest_held_title_tier >= tier_duchy
				#Check, must have some money
			}
			highest_held_title_tier <= tier_county
			trigger_if = {
				limit = {
					is_landed = yes
				}
				NOR = {
					government_has_flag = government_is_republic
					government_has_flag = government_is_theocracy
				}
			}
			any_held_title = { count = 0 }
		}
	}

	NOR = {
		government_has_flag = government_is_republic
		government_has_flag = government_is_theocracy
	}
}
