on_game_start = {
	on_actions = {
		khd_fixes
		#khd_debug
	}
}

khd_fixes = {
	effect = {
		#Fix Zoroastrianism double doctrines
		religion:zoroastrianism_religion ?= {
			every_faith = {
				if = {
					limit = {
						has_doctrine = doctrine_monogamy
						has_doctrine = doctrine_concubines
					}
					remove_doctrine = doctrine_monogamy
				}
			}
		}
	}
}
