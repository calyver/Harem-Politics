# The Silkworm Goddess Ceremony
silkworm_goddess_ceremony_decision = {
	title = silkworm_goddess_ceremony_decision
	desc = silkworm_goddess_ceremony_decision_desc
	selection_tooltip = silkworm_goddess_ceremony_decision_tooltip
	picture = {
		reference = "gfx/interface/illustrations/decisions/decision_personal_religious.dds"
	}
	sort_order = 162
	cooldown = { years = 3 }
	confirm_text = silkworm_goddess_ceremony_decision_confirm

	is_shown = {
		is_male = yes
		can_have_four_consorts_trigger = yes
		OR = {
			primary_spouse ?= {
				is_female = yes
				is_available_healthy_adult = yes
				is_courtier_of = root
			}
			mother ?= {
				is_available_healthy_adult = yes
				is_courtier_of = root
			}
		}
	}
	
	is_valid = {
		OR = {
			primary_spouse ?= {
				custom_description = {
					text = is_head_of_the_harem_trigger_tooltip
					is_head_of_the_harem_trigger = {
						LIEGE = root
					}
				}
				is_available_healthy_adult = yes
				is_female = yes
			}
			mother ?= {
				is_available_healthy_adult = yes
				is_courtier_of = root
			}
		}
	}

	is_valid_showing_failures_only = {
		custom_tooltip = {
			text = mourning_bride_show
			NOT = { has_character_modifier = mourner_modifier }
		}

		custom_tooltip = {
			text = contagious_disease_bride_show
			has_epidemic_disease_trigger = no
		}
		custom_tooltip = {
			text = epidemic_bride_show
			NOR = {
				realm_has_any_epidemic = {
					SIZE = 1
					INTENSITY = minor
				}
				capital_county ?= { has_county_modifier = isolate_capital_decision_modifier }
			}
		}
		custom_tooltip = {
			text = seclusion_bride_show
			NOR = {
				has_character_modifier = isolating_modifier
			}
		}
	}

	effect = {
		save_scope_as = ruler_scope
		hidden_effect = {
			if = {
				limit = {
					any_courtier = {
						is_head_of_the_harem_trigger = { LIEGE = root }
						is_available_healthy_adult = yes
						is_courtier_of = root
						is_female = yes
						OR = {
							is_parent_of = scope:ruler_scope
							scope:ruler_scope.primary_spouse ?= this
						}
					}
				}
				random_courtier = {
					limit = {
						is_head_of_the_harem_trigger = { LIEGE = root }
					}
					save_scope_as = ritual_leader
				}
			}
			else_if = {
				limit = {
					mother ?= {
						is_available_healthy_adult = yes
						is_courtier_of = root
					}
				}
				mother = { save_scope_as = ritual_leader }
			}
			else_if = {
				limit = {
					primary_spouse ?= {
						is_female = yes
						is_available_healthy_adult = yes
						is_courtier_of = root
					}
				}
				primary_spouse = {
					save_scope_as = ritual_leader
				}
			}
		}
		trigger_event = {
			id = harem_events.2000
			days = { 8 16 }
		}
	}
	
	ai_potential = { always = yes }

	ai_check_interval = 36
	ai_will_do = {
		base = 30
		ai_value_modifier = {
			ai_zeal = 0.25
		}
		modifier = {
			is_valid_for_legitimacy_change = yes
			legitimacy_level < 6
			add = 5
		}
		modifier = {
			is_valid_for_legitimacy_change = yes
			highest_held_title_tier = tier_empire
			legitimacy_level < 6
			add = 5
		}
		modifier = {
			is_valid_for_legitimacy_change = yes
			legitimacy_level < 5
			add = 5
		}
		modifier = {
			is_valid_for_legitimacy_change = yes
			legitimacy_level < 4
			add = 5
		}
		modifier = {
			any_courtier = {
				is_head_of_the_harem_trigger = { LIEGE = root }
				is_available_healthy_adult = yes
				is_courtier_of = root
				is_female = yes
				OR = {
					is_parent_of = root
					root.primary_spouse ?= this
				}
				learning >= decent_skill_rating
			}
			add = 15
		}
	}
}
