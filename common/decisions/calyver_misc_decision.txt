## Bride Show for most rulers ##
# Command Family to attend Exams
worship_ancestors_decision = {
	decision_group_type = admin
	title = worship_ancestors_decision
	desc = worship_ancestors_decision_desc
	selection_tooltip = worship_ancestors_decision_tooltip
	picture = {
		reference = "gfx/interface/illustrations/decisions/decision_personal_religious.dds"
	}
	sort_order = 163
	cooldown = { years = 3 }

	is_shown = {
		OR = {
			culture = {
				OR = {
					has_cultural_parameter = dishu_system
					has_cultural_tradition = tradition_mystical_ancestors
				}
			}
			faith = {
				OR = {
					has_doctrine_parameter = parents_get_hooks_on_children
					has_doctrine = tenet_ancestor_worship
				}
			}
		}
		is_landed_or_landless_administrative = yes
	}
	
	is_valid = {
		# The ancestors don't like illegititimate descendants
		is_bastard_child_trigger = no
		is_faith_dominant_gender = yes
	}

	is_valid_showing_failures_only = {
		is_available_adult = yes
	}
	
	widget = {
		gui = "decision_view_widget_option_list_generic"
		controller = decision_option_list_controller
		decision_to_second_step_button = "SELECT_ACTION"
		show_from_start = yes

		item = { # Big offering
			value = big_offering_decision
			is_shown = {
				always = yes
			}
			is_valid = {
				gold >= medium_gold_value
			}
			current_description = big_offering_decision_desc
			localization = big_offering_decision
			icon = "gfx/interface/icons/message_feed/money.dds"
			ai_chance = { value = 60 }
		}
		item = { # Small offering
			value = small_offering_decision
			is_shown = {
				always = yes
			}
			is_valid = {
				gold >= minor_gold_value
			}
			current_description = small_offering_decision_desc
			localization = small_offering_decision
			icon = "gfx/interface/icons/message_feed/money.dds"
			ai_chance = { value = 100 }
		}
	}

	effect = {
		save_scope_as = ruler_scope
		hidden_effect = {
			if = {
				limit = {
					exists = scope:big_offering_decision
				}
				remove_short_term_gold = medium_gold_value
			}
			else = {
				remove_short_term_gold = minor_gold_value
			}
		}
		trigger_event = {
			id = harem_events.1000
			days = { 7 10 }
		}
	}
	
	ai_potential = { always = yes }

	ai_check_interval_by_tier = {
		barony = 0
		county = 96
		duchy = 48
		kingdom = 36
		empire = 36
		hegemony = 36
	}
	ai_will_do = {
		base = 30
		ai_value_modifier = {
			ai_zeal = 0.25
		}
		ai_value_modifier = {
			ai_greed = -0.2
		}
		modifier = {
			any_consort = {
				is_visibly_fertile = yes
				count > 0
			}
			NOT = {
				any_child = {
					is_bastard_child_trigger = no
					NOT = {
						has_trait = child_of_concubine
					}
					sex_same_as = root
				}
			}
			add = 20
		}
		modifier = {
			health <= poor_health
			add = 30
		}
		modifier = {
			has_tgp_dlc_trigger = yes
			government_allows = merit
			NOR = {
				top_liege = root 
				has_character_modifier = tgp_local_exam_initial_success
			}
			add = 20
		}
		modifier = {
			is_valid_for_legitimacy_change = yes
			add = 20
		}
		modifier = {
			is_at_war = yes
			add = 30
		}
	}
}
