is_eligible_harem_member_trigger = {
	is_physically_able_adult = yes
	court_owner ?= $LIEGE$
	OR = {
		is_consort_of = $LIEGE$
		AND = {
			any_consort = {
				count = 0
			}
			# Check the legitimacy of the family line
			OR = {
				# Grandparents
				AND = {
					is_grandparent_of = $LIEGE$
					trigger_if = { # Paternal/maternal grandparents
						limit = {
							$LIEGE$.faith = {
								NOT = { has_doctrine = doctrine_gender_equal }
							}
						}
						any_child = {
							is_parent_of = $LIEGE$
							trigger_if = {
								limit = {
									$LIEGE$.faith = {
										has_doctrine = doctrine_gender_male_dominated
									}
								}
								is_male = yes
							}
							trigger_else = {
								is_male = no
							}
						}
					}
				}
				# The mother/father
				is_parent_of = $LIEGE$
				AND = { # Paternal/maternal aunts/uncles
					is_uncle_or_aunt_of = $LIEGE$
					any_sibling = {
						#Check sex based on religious doctrines
						trigger_if = {
							limit = {
								$LIEGE$.faith = {
									NOT = { has_doctrine = doctrine_gender_equal }
								}
							}
							trigger_if = {
								limit = {
									$LIEGE$.faith = {
										has_doctrine = doctrine_gender_male_dominated
									}
								}
								is_male = yes
							}
							trigger_else = {
								is_male = no
							}
						}
						is_parent_of = $LIEGE$
					}
				}
				# Siblings
				is_sibling_of = $LIEGE$
			}
			# Check their gender based on religious doctrines
			trigger_if = {
				limit = {
					$LIEGE$.faith = {
						NOT = { has_doctrine = doctrine_gender_equal }
					}
				}
				trigger_if = {
					limit = {
						$LIEGE$.faith = {
							has_doctrine = doctrine_gender_male_dominated
						}
					}
					is_male = no
				}
				trigger_else_if = {
					limit = {
						$LIEGE$.faith = {
							has_doctrine = doctrine_gender_female_dominated
						}
					}
					is_male = yes
				}
				trigger_else = {
					always = no
				}
			}
		}
	}
}
is_active_harem_member_trigger = {
	is_eligible_harem_member_trigger = {
		LIEGE = $LIEGE$
	}
	is_imprisoned = no
	trigger_if = {
		limit = {
			is_consort_of = $LIEGE$
		}
		always = yes
	}
	trigger_else_if = {
		limit = {
			$LIEGE$.religion = religion:islam_religion
		}
		always = yes
	}
	trigger_else_if = {
		limit = {
			$LIEGE$ = {
				hp_has_seniority_based_harem_relatives_trigger = yes
			}
		}
		trigger_if = {
			limit = {
				$LIEGE$ = {
					can_have_imperial_consorts_trigger = no
				}
			}
			OR = {
				is_parent_of = $LIEGE$
				is_grandparent_of = $LIEGE$
			}
		}
		trigger_else = {
			always = no
		}
	}
	trigger_else = {
		always = no
	}
}

grand_consort_validity_trigger = {
	base_court_position_validity_trigger = { EMPLOYER = $EMPLOYER$ }
	is_landed = no
	custom_tooltip = {
		text = grand_consort_valid_character_tooltip
		trigger_if = {
				limit = {
					$EMPLOYER$ = {
						religion = religion:islam_religion
						primary_spouse ?= {
							is_courtier_of = $EMPLOYER$
							NOT = {
								is_head_of_the_harem_trigger = {
									LIEGE = $EMPLOYER$
								}
							}
						}
					}
				}
				this = $EMPLOYER$.primary_spouse
			}
			trigger_else_if = {
				limit = {
					$EMPLOYER$ = { hp_accepts_polygamy = yes }
				}
				is_spouse_of = $EMPLOYER$
				NOT = { this = $EMPLOYER$.primary_spouse }
			}
			trigger_else = {
				is_concubine_of = $EMPLOYER$
			}
	}
	custom_tooltip = {
		text = none_does_not_have_a_full_time_job
		trigger_if = {
			limit = { NOT = { has_court_position = grand_consort_court_position } }
			NOT = {
				has_any_court_position = yes
			}
		}
	}
}

calyver_rank_is_superior_trigger = {
	# $AUTHORITY$ = is the authority character
	# $TARGET_CONSORT$ = is the target consort
	# $HAREM_LIEGE$ = is the harem liege
	$AUTHORITY$ = {
		this != $TARGET_CONSORT$
		OR = {
			#Liege's parent
			is_grandparent_of = $HAREM_LIEGE$
			is_parent_of = $HAREM_LIEGE$
			#The PS
			$HAREM_LIEGE$.primary_spouse ?= this
			#The Four Consorts
			AND = {
				is_high_consort_trigger = {
					LIEGE = $HAREM_LIEGE$
				}
				$TARGET_CONSORT$ = {
					trigger_if = {
						limit = {
							exists = $HAREM_LIEGE$.primary_spouse
						}
						this != $HAREM_LIEGE$.primary_spouse
					}
					NOT = {
						is_high_consort_trigger = {
							LIEGE = $HAREM_LIEGE$
						}
					}
				}
			}
			#Spouses
			AND = {
				is_spouse_of = $HAREM_LIEGE$
				$TARGET_CONSORT$ = {
					is_concubine_of = $HAREM_LIEGE$
				}
			}
		}
	}
	$TARGET_CONSORT$ = {
		is_consort_of = $HAREM_LIEGE$
		NOR = {
			is_parent_of = $HAREM_LIEGE$
			is_grandparent_of = $HAREM_LIEGE$
		}
	}
}
is_head_of_the_harem_trigger = {
	# Regular concubines are excluded
	NOT = {
		is_regular_concubine_of = {
			LIEGE = $LIEGE$
		}
	}
	trigger_if = { # Do we have active members?
		limit = {
			$LIEGE$ = {
				any_courtier = {
					is_active_harem_member_trigger = {
						LIEGE = $LIEGE$
					}
					NOT = {
						is_regular_concubine_of = {
							LIEGE = $LIEGE$
						}
					}
				}
			}
		}
		is_active_harem_member_trigger = {
			LIEGE = $LIEGE$
		}
		NOT = {
			is_regular_concubine_of = {
				LIEGE = $LIEGE$
			}
		}
		# Muslim hierarchy - elder relatives rank higher
		trigger_if = { #Liege's mother is always the head of the harem
			limit = {
				$LIEGE$ = {
					religion = religion:islam_religion
					any_parent = {
						is_active_harem_member_trigger = {
							LIEGE = $LIEGE$
						}
						NOT = { is_consort_of = $LIEGE$ }
					}
				}
			}
			is_parent_of = $LIEGE$
			NOT = { is_consort_of = $LIEGE$ }
		}
		trigger_else_if = { #Grandparents
			limit = {
				$LIEGE$ = {
					any_ancestor = {
						is_grandparent_of = $LIEGE$
						is_active_harem_member_trigger = {
							LIEGE = $LIEGE$
						}
						NOT = { is_consort_of = $LIEGE$ }
					}
				}
			}
			is_grandparent_of = $LIEGE$
			NOT = { is_consort_of = $LIEGE$ }
		}
		trigger_else_if = { #Liege's mother is always the head of the harem
			limit = {
				$LIEGE$ = {
					any_parent = {
						is_active_harem_member_trigger = {
							LIEGE = $LIEGE$
						}
						NOT = { is_consort_of = $LIEGE$ }
					}
				}
			}
			is_parent_of = $LIEGE$
			NOT = { is_consort_of = $LIEGE$ }
		}
		trigger_else_if = { #Unmarried paternal aunt
			limit = {
				$LIEGE$ = {
					any_parent = {
						any_sibling = {
							is_active_harem_member_trigger = {
								LIEGE = $LIEGE$
							}
							NOT = { is_consort_of = $LIEGE$ }
						}
					}
				}
			}
			is_uncle_or_aunt_of = $LIEGE$
			NOT = { is_consort_of = $LIEGE$ }
		}
		trigger_else_if = { #Unmarried older sisters
			limit = {
				$LIEGE$ = {
					any_sibling = {
						age > $LIEGE$.age
						is_active_harem_member_trigger = {
							LIEGE = $LIEGE$
						}
						NOT = { is_consort_of = $LIEGE$ }
					}
				}
			}
			age > $LIEGE$.age
			NOT = { is_consort_of = $LIEGE$ }
		}
		# Consorts
		trigger_else_if = { #The primary spouse
			limit = {
				$LIEGE$.primary_spouse ?= {
					is_courtier_of = $LIEGE$
				}
			}
			this = $LIEGE$.primary_spouse
		}
		trigger_else_if = { #The grand consort
			limit = {
				$LIEGE$ = {
					employs_court_position = grand_consort_court_position
				}
			}
			has_court_position = grand_consort_court_position
		}
		trigger_else_if = { #High Consorts
			limit = {
				$LIEGE$ = {
					employs_high_consort_court_position_trigger = yes
				}
			}
			is_high_consort_trigger = {
				LIEGE = $LIEGE$
			}
		}
		trigger_else_if = { #Just any spouse
			limit = {
				$LIEGE$ = {
					any_spouse = {
						is_courtier_of = $LIEGE$
					}
				}
			}
			is_spouse_of = $LIEGE$
		}
		# Muslim hierarchy - younger relatives rank as fallback
		trigger_else_if = { #Younger sisters
			limit = {
				$LIEGE$ = {
					any_sibling = {
						is_active_harem_member_trigger = {
							LIEGE = $LIEGE$
						}
						NOT = { is_consort_of = $LIEGE$ }
					}
				}
			}
			is_sibling_of = $LIEGE$
			NOT = { is_consort_of = $LIEGE$ }
		}
		trigger_else = {
			always = no
		}
	}
	trigger_else_if = {
		limit = {
			$LIEGE$ = {
				any_ancestor = {
					OR = {
						is_parent_of = $LIEGE$
						is_grandparent_of = $LIEGE$
					}
					is_eligible_harem_member_trigger = {
						LIEGE = $LIEGE$
					}
					NOT = { is_consort_of = $LIEGE$ }
				}
			}
		}
		is_eligible_harem_member_trigger = {
			LIEGE = $LIEGE$
		}
		NOT = { is_consort_of = $LIEGE$ }
		trigger_if = {
			limit = {
				$LIEGE$ = {
					any_ancestor = {
						is_grandparent_of = $LIEGE$
						is_eligible_harem_member_trigger = {
							LIEGE = $LIEGE$
						}
						NOT = { is_consort_of = $LIEGE$ }
					}
				}
			}
			is_grandparent_of = $LIEGE$
		}
		trigger_else = {
			is_parent_of = $LIEGE$
		}
	}
	trigger_else = {
		always = no
	}
}

is_secondary_spouse_trigger = {
	is_married = yes
	save_temporary_scope_as = harem_spouse
	NOR = {
		has_court_position = grand_consort_court_position
		has_court_position = pure_consort_court_position
		has_court_position = virtuous_consort_court_position
		has_court_position = worthy_consort_court_position

			primary_spouse ?= scope:harem_spouse
		}
	}
}

	}

should_be_di_child_trigger = {
	is_bastard_child_trigger = no
	house ?= {
		any_house_member = {
			this = $PARENT$
		}
	}
	NOR = {
		has_trait = di_child
		has_trait = child_of_concubine
		any_parent = {
			NOT = { this = $PARENT$ }
			is_landed_or_landless_administrative = yes
		}
	}
	trigger_if = {
		limit = {
			any_parent = {
				NOT = { this = $PARENT$ }
				even_if_dead = yes
				count > 0
			}
		}
		any_parent = { # The parent was either divorced as PS, or is the PS
			NOT = { this = $PARENT$ }
			even_if_dead = yes
			OR = {
				$PARENT$.primary_spouse ?= this
				has_opinion_modifier = {
					target = $PARENT$
					modifier = divorced_me_as_ps
				}
				AND = {
					$PARENT$ = { hp_accepts_polygamy = no }
					any_former_spouse = {
						this = $PARENT$
					}
				}
			}
			NOR = {
				has_opinion_modifier = {
					target = $PARENT$
					modifier = dismissed_me_as_ps
				}
				has_opinion_modifier = {
					target = $PARENT$
					modifier = demoted_me_as_ps
				}
			}
		}
	}
}
