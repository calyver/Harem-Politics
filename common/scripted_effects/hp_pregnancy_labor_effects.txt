calyver_remove_pregnancy_stability_modifiers_effect = {
	if = {
		limit = {
			has_character_modifier = super_stable_pregnancy_modifier
		}
		remove_character_modifier = super_stable_pregnancy_modifier
	}
	if = {
		limit = {
			has_character_modifier = unstable_pregnancy_modifier
		}
		remove_character_modifier = unstable_pregnancy_modifier
	}
}
calyver_add_super_stable_pregnancy_modifier_effect = {
	if = {
		limit = {
			has_character_modifier = unstable_pregnancy_modifier
		}
		remove_character_modifier = unstable_pregnancy_modifier
	}
	if = {
		limit = {
			NOT = {
				has_character_modifier = super_stable_pregnancy_modifier
			}
		}
		add_character_modifier = {
			modifier = super_stable_pregnancy_modifier
		}
	}
}
calyver_add_unstable_pregnancy_modifier_effect = {
	if = {
		limit = {
			has_character_modifier = super_stable_pregnancy_modifier
		}
		remove_character_modifier = super_stable_pregnancy_modifier
	}
	if = {
		limit = {
			NOT = {
				has_character_modifier = unstable_pregnancy_modifier
			}
		}
		add_character_modifier = {
			modifier = unstable_pregnancy_modifier
		}
	}
}
