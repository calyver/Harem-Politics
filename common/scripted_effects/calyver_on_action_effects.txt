## Dishu System ##
assign_di_child_trait_to_all_children_effect = {
	if = {
		limit = {
			is_dishu_parent_trigger = yes
		}
		save_scope_as = dishu_parent
		every_child = {
			limit = {
				should_be_di_child_trigger = {
					PARENT = scope:dishu_parent
				}
			}
			add_trait = di_child
		}
	}
}
