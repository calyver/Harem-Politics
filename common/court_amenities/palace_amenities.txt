		ai_will_do = {
			value = 60
			if = {
				limit = {
					has_personality_extroverted_trigger = yes

				limit = {
					has_personality_dominant_trigger = yes
				}
				add = 3
			}

				}

			}
			if = { # Prefer over fashion

					ai_rationality > 0
				}
				add = 1
			}
			if = {
				limit = {
					government_has_flag = government_is_tribal
				}
				add = -1000
			}
