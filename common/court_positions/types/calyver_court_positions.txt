keeper_of_the_harem_court_position = {
	sort_order = {
		value = 370
		if = {
			limit = {
				primary_title = title:h_china
				culture = {
					has_cultural_pillar = heritage_chinese
				}
			}
			add = 640
		}
	}
	max_available_positions = 1
	minimum_rank = kingdom # Should only be available for kings and above
	skill = diplomacy

	court_position_asset = {
		trigger = { culture_has_east_asian_heritage_pillar_trigger = yes }
		animation = fanning_coyly
		background = "gfx/interface/illustrations/event_scenes/tgp_wedding_bedroom_asia.dds"
	}

	court_position_asset = {
		animation = survey
		background = "gfx/interface/illustrations/event_scenes/corridor.dds"
	}

	opinion = {
		value = regular_court_position_opinion
	}

	# How high the aptitude has to be to reach the 5 different modifier levels
	aptitude_level_breakpoints = { 20 40 60 80 }

	# How good is this character in this position? root scope is the holder of the court position
	aptitude = {
		value = 1
		add = {
			value = diplomacy
			multiply = 0.5
			max = 50
			desc = court_position_skill_diplomacy
		}
		add = {
			value = intrigue
			multiply = 0.5
			max = 50
			desc = court_position_skill_intrigue
		}
		if = {
			limit = {
				has_trait = diligent
			}
			add = {
				value = 10
				desc = court_position_diligent_trait
			}
		}
		if = {
			limit = {
				has_trait = patient
			}
			add = {
				value = 5
				desc = court_position_patient_trait
			}
		}
		if = {
			limit = {
				has_trait = calm
			}
			add = {
				value = 5
				desc = court_position_calm_trait
			}
		}
		if = {
			limit = {
				has_trait = intellect_good_3
			}
			add = {
				value = 15
				desc = court_position_intellect_good_3_trait
			}
		}
		else_if = {
			limit = {
				has_trait = intellect_good_2
			}
			add = {
				value = 10
				desc = court_position_intellect_good_2_trait
			}
		}
		else_if = {
			limit = {
				has_trait = intellect_good_1
			}
			add = {
				value = 5
				desc = court_position_intellect_good_1_trait
			}
		}
		if = {
			limit = {
				has_trait = shrewd
			}
			add = {
				value = 10
				desc = court_position_shrewd_trait
			}
		}
		if = {
			limit = {
				has_trait = reclusive
			}
			add = {
				value = -10
				desc = court_position_reclusive_trait
			}
		}
		if = {
			limit = {
				has_trait = shy
			}
			add = {
				value = -10
				desc = court_position_shy_trait
			}
		}
		if = {
			limit = {
				has_trait = lazy
			}
			add = {
				value = -10
				desc = court_position_lazy_trait
			}
		}

		if = {
			limit = {
				has_no_particular_noble_roots_trigger = yes
			}

				value = 30
				desc = has_no_particular_noble_roots_trigger
			}

			limit = {

	# Is this an available position for this court? (root is court owner)
	is_shown = {

			AND = {
				is_male = yes
				OR = {
					has_realm_law = male_only_law
					has_realm_law = male_preference_law
				}
			}
			AND = {
				is_female = yes
				OR = {
					has_realm_law = female_only_law 
					has_realm_law = female_preference_law
				}
			}
		}
	}

		scope:employee = {
			OR = {
				is_eunuch_trigger = yes
				AND = {
					is_female = yes

						OR = {
							has_realm_law = male_only_law
							has_realm_law = male_preference_law
						}
					}
				}

					is_female = no

						OR = {
							has_realm_law = female_only_law 
							has_realm_law = female_preference_law
						}
					}
				}
			}
			NOT = {
				is_consort_of = scope:liege
			}
			court_position_employee_shown_trigger = yes
		}
	}

	valid_character = {
		scope:employee = {
			keeper_of_the_harem_validity_trigger = { EMPLOYER = scope:liege }
		}
	}

	revoke_cost = { # root is the liege, no other scopes are passed here!
		prestige =  {
			value = major_court_position_prestige_revoke_cost
			# Set to 0 via obligation hook
			if = {
				limit = {
					court_position_obligation_hook_salary_trigger = {
						POS = keeper_of_the_harem
						LIEGE = root
					}
				}
				multiply = 0
			}
			# Avoid AI prestige issues
			if = {
				limit = {
					exists = this
					is_ai = yes
				}
				multiply = 0
			}
			# Special case for events
			if = {
				limit = {
					exists = this
					has_character_flag = free_revoke_cost
				}
				multiply = 0
			}
		}
	}

	salary = {
		gold = {
			value = 0
			if = {
				limit = {
					scope:liege = {
						has_treasury = no
					}
				}
				add = keeper_of_the_harem_total_salary_value
			}
		}
		treasury = {
			value = 0
			if = {
				limit = {
					scope:liege = {
						has_treasury = yes
					}
				}
				add = keeper_of_the_harem_total_salary_value
			}
		}
		prestige = {
			value = minor_court_position_prestige_salary
		}
		round = no
	}

	received_salary = {
		gold = {
			value = keeper_of_the_harem_total_salary_value
		}
		prestige = {
			value = minor_court_position_prestige_salary
		}
		round = no
	}

	base_employer_modifier = {
	}

	modifier = {
	}

	is_powerful_agent = yes

	# These modifiers are employed based on what level the aptitude reaches (levels are defined as breakpoints)
	scaling_employer_modifiers = {
		terrible = {
			spouse_opinion = 5
		}
		poor = {
			spouse_opinion = 10
		}
		average = {
			spouse_opinion = 15
		}
		good = {
			spouse_opinion = 25
		}
		excellent = {
			spouse_opinion = 40
		}
	}

	on_court_position_received = {
		keeper_of_the_harem_title_accepted_effect = yes
	}
	on_court_position_revoked = {
		keeper_of_the_harem_title_revoked_effect = yes
	}
	on_court_position_invalidated = {
		keeper_of_the_harem_title_invalidated_effect = yes
	}
	on_court_position_vacated = {
		court_position_vacated_event_effect = {
			COURT_POSITION_TYPE = keeper_of_the_harem_court_position
		}
	}
	# Scripted value to determine whether an AI will hire/fire a position or not. Hires over 0, fires under -50. Look in _court_positions.info for more extensive documentation!
	ai_position_score = {
		value = 100
		if = {
			limit = {
				scope:liege = {
					any_consort = {
						is_valid_for_favor_system_trigger = { HAREM_LIEGE = scope:liege }
						count > 4
					}
				}
			}
			add = 50
		}
		if = {
			limit = {
				scope:liege = {
					any_consort = {
						is_valid_for_favor_system_trigger = { HAREM_LIEGE = scope:liege }
						count > 5
					}
				}
			}
			add = 50
		}
		if = {
			limit = {
				scope:liege.highest_held_title_tier >= tier_empire
			}
			add = 50
		}
		if = {
			limit = {
				scope:liege.highest_held_title_tier > tier_empire
			}
			add = 50
		}
		add = court_position_debt_considerations_value
	}
	ai_candidate_score = {
		value = 50
		add = scope:base_value
		add = court_position_candidate_score_base_value
		add = court_position_candidate_aptitude_value
		if = { # To stop petitioned positions immediately being fired by the AI
			limit = {
				court_position_petition_ai_weight_trigger = { ROLE = fire_dragon_engineer }
			}
			add = 1000
		}
	}
}

									is_courtier_of = scope:liege
									is_vassal_of = scope:liege
								}
							}
						}
					}
				}
				add = 10
			}
			# Big harems need help in management
			if = {
				limit = {
					any_consort = {
						is_valid_for_favor_system_trigger = { HAREM_LIEGE = scope:liege }
						count >= 6
					}
				}
				add = {
					value = 15
					if = {
						limit = {
							any_consort = {
								is_valid_for_favor_system_trigger = { HAREM_LIEGE = scope:liege }
								count >= 8
							}
						}
						add = 10
					}
				}
			}
			# Noble birth
			if = {
				limit = {
					scope:employee = {
						is_lowborn = yes
					}
				}
				add = -10
			}
			if = {
				limit = {
					scope:employee = { #We don't want to make a lowborn the PS
						is_lowborn = yes
						is_spouse_of = scope:liege
					}
				}
				add = -10
			}
			# Powerful relatives
			if = {
				limit = {
					scope:employee = {
						any_close_family_member = {
							NOT = { is_child_of = scope:liege }
							is_allied_to = scope:liege
						}
					}
				}
				add = 10
			}
			if = {
				limit = {
					scope:employee = {
						any_close_family_member = {
							is_vassal_of = scope:liege
							NOT = { is_child_of = scope:liege }
						}
					}
				}
				add = 10
			}
			if = {
				limit = {
					scope:employee = {
						any_close_family_member = {
							OR = {
								is_powerful_vassal_of = scope:liege
								is_councillor_of = scope:liege
							}
							NOT = { is_child_of = scope:liege }
						}
					}
				}
				add = 25
			}
			if = {
				limit = {
					scope:employee = {
						any_close_family_member = {
							is_liege_or_above_of = scope:liege
							NOT = { is_child_of = scope:liege }
						}
					}
				}
				add = 50
			}
			if = {
				limit = {
					scope:employee = {
						any_close_family_member = {
							tier_difference = {
								target = scope:liege
								value >= -2
							}
							NOT = { is_child_of = scope:liege }
						}
					}
				}
				add = {
					value = 5
					if = {
						limit = {
							scope:employee = {
								any_close_family_member = {
									tier_difference = {
										target = scope:liege
										value >= -1
									}
									NOT = { is_child_of = scope:liege }
								}
							}
						}
						add = 5
					}
					if = {
						limit = {
							scope:employee = {
								any_close_family_member = {
									tier_difference = {
										target = scope:liege
										value >= 0
									}
									NOT = { is_child_of = scope:liege }
								}
							}
						}
						add = 10
					}
					if = {
						limit = {
							scope:employee = {
								any_close_family_member = {
									tier_difference = {
										target = scope:liege
										value >= 1
									}
									NOT = { is_child_of = scope:liege }
								}
							}
						}
						add = 30
					}
				}
			}
			# Children
			if = {
				limit = {
					any_child = {
						is_child_of = scope:employee
						is_bastard_child_trigger = no
						NOT = {
							has_trait = disinherited
						}
					}
				}
				every_child = {
					limit = {
						is_child_of = scope:employee
						is_bastard_child_trigger = no
						NOT = {
							has_trait = disinherited
						}
					}
					add = {
						value = 5
						if = {
							limit = {
								is_player_heir_of = scope:liege
							}
							add = 20
						}
						if = {
							limit = {
								character_gender_can_inherit_from_trigger = { CHARACTER = scope:liege }
							}
							add = 5
						}
					}
				}
			}
			# Relations
			if = {
				limit = {
					has_relation_soulmate = scope:employee
				}
				add = 30
			}
			if = {
				limit = {
					OR = {
						has_relation_lover = scope:employee
						has_relation_best_friend = scope:employee
					}
				}
				add = 15
			}
			else_if = {
				limit = {
					has_relation_friend = scope:employee
				}
				add = 15
			}
			if = {
				limit = {
					has_relation_rival = scope:employee
				}
				add = -20
			}
			else_if = {
				limit = {
					has_relation_nemesis = scope:employee
				}
				add = -35
			}
			# Favor
			if = {
				limit = {
					scope:employee = {
						harem_politics_beloved_consort_trigger = yes
					}
				}
				add = 15
			}
			if = {
				limit = {
					scope:employee = {
						harem_politics_favored_consort_trigger = yes
					}
				}
				add = 10
			}
			if = {
				limit = {
					scope:employee = {
						has_character_modifier = favor_boost_modifier
					}
				}
				add = 10
			}
			if = {
				limit = {
					scope:employee = {
						harem_politics_unfavored_consort_trigger = yes
					}
				}
				add = -10
			}
			if = {
				limit = {
					scope:employee = {
						harem_politics_forgotten_consort_trigger = yes
					}
				}
				add = -10
			}
			# Opinion
			add = {
				value = "scope:liege.opinion(scope:employee)"
				multiply = 0.15
			}
		}
		if = { # To stop petitioned positions immediately being fired by the AI
			limit = {
				court_position_petition_ai_weight_trigger = { ROLE = worthy_consort }
			}
			add = 1000
		}
	}
