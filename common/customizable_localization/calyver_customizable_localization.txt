consort_rank_localization = {
	type = character

	text = { #The PS
		trigger = {
			any_spouse = {
				primary_spouse = {
					this = root
				}
			}
		}
		localization_key = PRIMARY_SPOUSE
	}

	#GC
	text = {
		trigger = {
			has_court_position = grand_consort_court_position
		}
		localization_key = GRAND_CONSORT
	}

	#SS
	text = {
		trigger = {
			court_owner ?= {
				any_spouse = {
					this = root
				}
			}
		}
		localization_key = SECONDARY_SPOUSE_HAREM_WINDOW
	}

	text = { #The rest
		trigger = {
			court_owner ?= {
				root = {
					is_concubine_of = prev
				}
			}
		}
		localization_key = CONCUBINE_HAREM_WINDOW
	}

	# Relatives
	text = {
		trigger = {
			is_male = yes
		}
		localization_key = PATRON_HAREM_WINDOW
	}
	text = {
		localization_key = MATRON_HAREM_WINDOW
	}
}
